Parameters:
Name Type Argument Description
selector String widget css selector
template Object <optional>
 widget template

Source:
  • better-dom.js, line 1709
Example
// simple example
DOM.extend(".myplugin", {
    append: "<span>myplugin text</span>"
}, {
    constructor: function() {
        // initialize extension
    }
});

// emmet-like syntax example
DOM.extend(".mycalendar", {
    after: "table>(tr>th*7)+(tr>td*7)*6"
}, {
    constructor: function() {
        // initialize extension
    },
    method1: function() {
        // this method will be mixed into every instance
    },
    method2: function() {
        // this method will be mixed into evety instance
    }
});
